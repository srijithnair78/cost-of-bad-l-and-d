/ --- 1. THE DYNAMIC TEXTURE (Cost of Bad L&D) ---

function createReceiptTexture(gl) {

const canvas = document.createElement('canvas');

canvas.width = 512; canvas.height = 1024;

const ctx = canvas.getContext('2d');


// Receipt Paper Look

ctx.fillStyle = '#fdfdfd';

ctx.fillRect(0, 0, canvas.width, canvas.height);


// Text Setup

ctx.fillStyle = '#111';

ctx.textAlign = 'center';

ctx.font = 'bold 32px monospace';

ctx.fillText('THE COST OF BAD L&D', 256, 80);


ctx.font = '16px monospace';

ctx.textAlign = 'left';

const items = [

{ name: "Talent Disengaged", cost: "$82,000" },

{ name: "Skills Not Applied", cost: "$41,000" },

{ name: "Leadership Stalled", cost: "$36,000" },

{ name: "Wrong Learners", cost: "$29,000" }

];


items.forEach((item, i) => {

let y = 180 + (i * 100);

ctx.font = 'bold 20px monospace';

ctx.fillText(item.name, 40, y);

ctx.textAlign = 'right';

ctx.fillText(item.cost, 470, y);

ctx.textAlign = 'left';

ctx.font = '14px monospace';

ctx.fillText("Investment without impact.", 40, y + 25);

});


ctx.font = 'bold 28px monospace';

ctx.fillText('TOTAL: $300,000.00', 40, 850);


const tex = gl.createTexture();

gl.bindTexture(gl.TEXTURE_2D, tex);

gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, canvas);

gl.generateMipmap(gl.TEXTURE_2D);

return tex;

}


// --- 2. THE PHYSICS MESH (Verlet) ---

const mouse = { x: 0, y: 0, down: false };

window.addEventListener('mousemove', e => {

mouse.x = (e.clientX / window.innerWidth) * 2 - 1;

mouse.y = -(e.clientY / window.innerHeight) * 2 + 1;

});

window.addEventListener('mousedown', () => mouse.down = true);

window.addEventListener('mouseup', () => mouse.down = false);


function updatePhysics(points, constraints) {

points.forEach(p => {

if (p.pinned) return;


// Interaction: If mouse is near a point, pull it

if (mouse.down) {

let dx = mouse.x * 2 - p.x; // Scale mouse to mesh space

let dy = mouse.y * 2 - p.y;

let dist = Math.sqrt(dx*dx + dy*dy);

if (dist < 0.5) {

p.x += dx * 0.1;

p.y += dy * 0.1;

}

}


let vx = (p.x - p.oldX) * 0.96;

let vy = (p.y - p.oldY) * 0.96;

p.oldX = p.x; p.oldY = p.y;

p.x += vx; p.y += vy - 0.0015; // Gravity

});


// Spring satisfaction (Stiffness)

for (let i = 0; i < 8; i++) {

constraints.forEach(c => {

let p1 = points[c[0]], p2 = points[c[1]];

let dx = p2.x - p1.x, dy = p2.y - p1.y;

let dist = Math.sqrt(dx*dx + dy*dy);

let diff = (c[2] - dist) / dist * 0.5;

if (!p1.pinned) { p1.x -= dx * diff; p1.y -= dy * diff; }

if (!p2.pinned) { p2.x += dx * diff; p2.y += dy * diff; }

});

}

}
